	"#!/bin/sh\n"
	"exec java -server ${MAC_JAVA_FIX} -Xincgc -Xmx64m -Xms32m -ea -cp ${CMAKE_INSTALL_PREFIX}/share/java/lcm.jar lcm.logging.LogPlayer \$*\n")
